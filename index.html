Problem #1: Run Length Encoding


@given(st.lists(st.integers()))
def test_encoding_never_crashes(ls):
    assert run_length_decode(run_length_encode(ls)) == ls


@given(st.lists(st.integers()))
def test_encoding_never_increases_length(ls):
    assert len(run_length_encode(ls)) <= len(ls)


@given(st.lists(st.integers()), st.integers(0, 10))
def test_encode_a_constant(v, n):
    assert len(run_length_encode([v] * n)) == [(v, n)]

Problem #2: Binary search


@given(st.lists(st.integers()).map(sorted), st.integers())
def test_insert_is_sorted(ls, v):
    ls.insert(binary_search(ls, v), v)
    assert ls == sorted(ls)


@given(st.lists(st.integers()).map(sorted), st.integers())
def test_insert_earlier_is_not_sorted(ls, v):
    i = binary_search(ls, v)
    for j in range(i):
        ls2 = list(ls)
        ls2.insert(j, v)
        assert ls2 != sorted(ls2)


@given(st.lists(st.integers()).map(sorted), st.integers())
def test_insert_does_not_change_insert(ls, v):
    i = binary_search(ls, v)
    ls.insert(i, v)
    assert binary_search(ls, v) == i

Problem #3: Cron


@given(
  st.integers(0, 23) | st.none(), st.integers(0, 59) | st.none(), 
  st.integers(0, 23), st.integers(0, 59)) 
def test_bounds(hs, ms, ch, cm):
    h, m = cron(hs, ms, ch, cm)
    assert 0 <= h < 24
    assert 0 <= m < 60



@given(
  st.integers(0, 23) | st.none(), st.integers(0, 59) | st.none(), 
  st.integers(0, 23), st.integers(0, 59)) 
def test_answer_is_admissible(hs, ms, ch, cm):
    h, m = cron(hs, ms, ch, cm)
    if hs is not None:
        assert hs == h
    if ms is not None:
        assert ms == m


@given(
  st.integers(0, 23) | st.none(), st.integers(0, 59) | st.none(), 
  st.integers(0, 23), st.integers(0, 59)) 
def test_widening_produces_earlier_answer(hs, ms, h, m):
    when = cron(hs, ms, h, m)
    assume(when > (h, m))
    assert cron(None, ms, h, m) <= when
    assert cron(hs, None, h, m) <= when


@given(
  st.integers(0, 23) | st.none(), st.integers(0, 59) | st.none(), 
  st.integers(0, 23), st.integers(0, 59)) 
def test_agrees_with_brute_force_solution(hs, ms, h, m):
    if hour_spec is None:
        allowed_hours = range(24)
    else:
        allowed_hours = [hour_spec]
    if minute_spec is None:
        allowed_minutes = range(60)
    else:
        allowed_minutes = [minute_spec]
    times = [
        (h, m) for h in allowed_hours for m in allowed_minutes
    ]
    later_times = [

    else:

Problem #4: Flood fill

(Bonus Question)


@composite
def problems(draw):
    image = draw(images())
    i = draw(Indices)
    assume(i < image.width)
    j = draw(Indices)
    assume(j < image.height)
    colour = draw(Pixels)
    assume(image[i, j] != colour)
    return image, i, j, colour


@given(problems())
def test_flood_fill_fills_point(problem):
    image, i, j, colour = problem
    flood_fill(image, i, j, colour)
    assert image[i, j] == colour


@given(problems())
def test_flood_fill_clears_adjacent_points(problem):
    image, i, j, colour = problem
    m, n = image.size
    original_colour = image[i, j]
    flood_fill(grid, i, j, colour)
    for u, v in [
      (i - 1, j), (i + 1, j), (i, j - 1), (i, j + 1)
    ]:
        if 0 <= u < m and 0 <= v < n:
            assert grid[u][v] != original_colour


def transpose(grid):
    m, n = grid.size
    transposed = Image(n, m)
    for i in range(m):
        for j in range(n):
            transposed[j, i] = grid[i, j]
    return transposed

@given(problems())
def test_commutes_with_transposing(problem):
    grid, i, j, colour = problem
    grid2 = transpose(grid)

    flood_fill(grid, i, j, colour)
    flood_fill(grid2, j, i, colour)
    assert grid == transpose(grid2)


@given(problems())
def test_flood_filling_from_unfilled_does_not_return(problem):
    grid, i, j = problem
    original = deepcopy(grid)
    flood_fill(grid, i, j)
    for u in range(len(original)):
        for v in range(len(original[u])):
            if not grid[u][v]:
                newgrid = deepcopy(original)
                flood_fill(newgrid, u, v)
                assert not newgrid[i][j]


@given(problems(), data())
def test_reachability_is_symmetric(problem, data):
    image, i, j, colour = problem
    original = image[i, j]
    cloned = image.copy()
    flood_fill(cloned, i, j, colour)
    other_points = [
        (u, v)
        for u in range(image.width) for v in range(image.height)
        if cloned[u, v] == original
    ]
    assume(other_points)
    u, v = data.draw(sampled_from(other_points))
    flood_fill(image, u, v, colour)
    assert image[i, j] == original
